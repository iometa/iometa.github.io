基本操作

# 配置身份
git config --global user.name "用户名"
git config --global user.email "邮箱地址"
# 创建代码仓库(首先要进入项目的根目录)
git init
# 添加文件(添加要提交的文件)
git add filename # 添加单个文件
git add . # 添加目录下的所有文件
# 提交文件(提交代码到代码仓库)
git commit -m "commit message here"
# 查看修改的部分(代码提交后,本地又发生更改)
git status # 显示被修改的文件
git diff # 显示被修改的文件的具体被修改的内容
# 显示历史提交信息
git log
# 撤销未添加文件的更改
git checkout **/**/**.**
# 取消添加文件
git reset HEAD **/**/**.**
# 回退到上一个版本
git reset --hard HEAD
git reset --hard HEAD^
# 回退或恢复到指定版本
git reset --hard version_number
# 查询所有的历史版本更改信息
git reflog
# clone project
git clone ***
# tag a important version
git tag -a v1.0 version_number
# show tags
git tag
# add remote repository. Note: add a new repository on github first
git remote add shortname git@github.com:github_username/project_name.git
# config verification information
ssh-keygen -t rsa -C "Your github account email"
# test 
ssh -T git@github.com
# view the remote library
git remote -v
# fetch changes from remote library
git fetch
# merge changes
git merge
# 将本地更改推送到远程服务器上
git push
# some basic setting 
git branch --set-upstream-to=origin/master master
